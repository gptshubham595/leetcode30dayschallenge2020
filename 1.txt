class Solution {
public:
    int singleNumber(vector<int>& nums) {
        unordered_map<int,int> m;
        int sum=0;
        for(int i=0;i<nums.size();i++)sum+=nums[i];
        
        for(int i=0;i<nums.size();i++)
        {   if(m.count(nums[i])){
            m[nums[i]]=1; sum-=2*nums[i];
        }
            m[nums[i]]=0;
        }
        return sum;
       
    }
};
