class Solution {
public:
    bool canJump(vector<int>& nums) {
        ios_base::sync_with_stdio(false);
        cin.tie(nullptr);
        
         int maxPos=0, N=nums.size();

        for (int i=0;i<=maxPos && i <N-1 ; i++)
        {   
            maxPos=max(i + nums[i], maxPos);
        }

        return maxPos >=N-1; 
        
    }
};
