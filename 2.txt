class Solution {
public:
    int sum(int n){
        int t=n; 
        int s=0;
        while(t>0){
            int r=t%10;
            s+=r*r;
            t=t/10;
        }
        return s;
    }
    bool isHappy(int n) {
        int slow, fast;
        slow = fast = n;
        do {
            slow = sum(slow);
            fast = sum(fast);
            fast = sum(fast);
        } while(slow != fast);
        if (slow == 1) return 1;
        else return 0;
    }
};
