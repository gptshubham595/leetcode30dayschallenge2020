class Solution {
public:
    void moveZeroes(vector<int>& n) {
        vector<int> b;
        int z=0;
        for(int i=0;i<n.size();i++){
            if(n[i]==0){
                z++;
            }else{b.push_back(n[i]);}
        }
        for(int i=0;i<z;i++){
            b.push_back(0);
        }
        n=b;
    }
};
